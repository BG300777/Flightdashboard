    <script>
        // Update current time
        function updateCurrentTime() {
            const now = new Date();
            const options = { 
                day: '2-digit', 
                month: 'short', 
                year: 'numeric', 
                hour: '2-digit', 
                minute: '2-digit', 
                second: '2-digit',
                hour12: false 
            };
            const formattedTime = now.toLocaleDateString('en-GB', options).replace(',', ' â€¢');
            document.getElementById('current-time').textContent = formattedTime;
        }
        
        // Update flight statuses randomly to simulate real-time changes
        function updateFlightStatuses() {
            const statusCells = document.querySelectorAll('.status');
            const timeCells = document.querySelectorAll('tbody tr td:nth-child(4)');
            const now = new Date();
            
            statusCells.forEach((cell, index) => {
                // Only update some statuses randomly (not all at once)
                if (Math.random() > 0.7) {
                    const statusTypes = ['departed', 'boarding', 'not-open'];
                    const statusIcons = {
                        'departed': 'âœ…',
                        'boarding': 'ðŸ’¬',
                        'not-open': 'âœ…'
                    };
                    
                    // Remove all status classes
                    cell.classList.remove('departed', 'boarding', 'not-open');
                    
                    // Select a random status
                    const randomStatus = statusTypes[Math.floor(Math.random() * statusTypes.length)];
                    
                    // Apply new status
                    cell.classList.add(randomStatus);
                    cell.innerHTML = `<span class="status-icon">${statusIcons[randomStatus]}</span> ${randomStatus.charAt(0).toUpperCase() + randomStatus.slice(1).replace('-', ' ')}`;
                    
                    // Update the time for the changed status
                    const timeOptions = { 
                        day: '2-digit', 
                        month: 'short', 
                        year: 'numeric', 
                        hour: '2-digit', 
                        minute: '2-digit', 
                        second: '2-digit',
                        hour12: false 
                    };
                    timeCells[index].textContent = now.toLocaleDateString('en-GB', timeOptions).replace(',', ' â€¢');
                }
            });
            
            // Update last updated time
            const lastUpdated = now.toLocaleDateString('en-GB', { 
                day: '2-digit', 
                month: 'short', 
                year: 'numeric', 
                hour: '2-digit', 
                minute: '2-digit', 
                second: '2-digit',
                hour12: false 
            }).replace(',', ' â€¢');
            document.getElementById('last-updated').textContent = `Last updated: ${lastUpdated}`;
        }
        
        // Initial setup
        updateCurrentTime();
        
        // Set interval to update time every second
        setInterval(updateCurrentTime, 1000);
        
        // Set interval to update flight statuses every 60 seconds if auto-refresh is enabled
        setInterval(() => {
            if (document.getElementById('auto-refresh-check').checked) {
                updateFlightStatuses();
            }
        }, 60000);
        
        // Add event listener for manual refresh button
        document.getElementById('refresh-btn').addEventListener('click', updateFlightStatuses);
        
        // Initial status update
        setTimeout(updateFlightStatuses, 2000);
    </script>
